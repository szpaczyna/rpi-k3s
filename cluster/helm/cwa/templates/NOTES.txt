  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "cwa.name" . }},app.kubernetes.io/instance={{ .Release.Name }},app.kubernetes.io/component=main" -o jsonpath="{.items[0].metadata.name}")
  echo "Visit http://127.0.0.1:{{ .Values.components.calibreWebAutomated.service.port }} to use your application"

2. To access the book downloader service:
  export DOWNLOADER_POD=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "cwa.name" . }},app.kubernetes.io/instance={{ .Release.Name }},app.kubernetes.io/component=downloader" -o jsonpath="{.items[0].metadata.name}")
